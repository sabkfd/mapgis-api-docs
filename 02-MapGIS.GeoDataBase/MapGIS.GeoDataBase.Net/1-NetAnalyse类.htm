DeleliverAnalyse 方法示例

C#

 
using MapGIS.GeoDataBase.Net;
using MapGIS.GeoDataBase;
 
//定义变量
DataBase GDB = null;
Server _GDBServer = null;
NetAnalyse _NetAnalyse = null;
NetCls _NetCls = null;
//出发地网标
NetElement StaritemElement= null;
List<NetElement> StartElements = null;
//目的地网标
NetElement ToitemElement = null;
List<NetElement> ToElements = null;
List<MDeliverPathInfo> deliverInfos=null;
//连接数据源，打开数据库          
_GDBServer = new Server();
bool rtn = _GDBServer.Connect("MapGisLocal", "", "");
if (rtn)
{
    GDB = _GDBServer.OpenGDB("DevelopDB");
    //打开网络类
    _NetCls = new NetCls(GDB);
    _NetCls.Open("net", 0);
    //变量初始化
    _NetAnalyse = new NetAnalyse();
    StartElements = new List<NetElement>();
    ToElements = new List<NetElement>();
    deliverInfos=new List<MDeliverPathInfo>();
    //设置起点网标信息
    StaritemElement = new NetElement();
    StaritemElement.ElemID = 2795;
    StaritemElement.Type = ElemType.NodeElem;
    StartElements.Add(StaritemElement);
    //设置目的地网标
    ToitemElement = new NetElement();
    ToitemElement.ElemID = 3418;
    ToitemElement.Type = ElemType.NodeElem;
    ToElements.Add(ToitemElement);
    //设置网络类
    _NetAnalyse.Cls = _NetCls;
    //多车配送分析，设置起点、目的点信息，返回分析结果
    rtn = _NetAnalyse.DeleliverAnalyse( StartElements,ToElements,out deliverInfos );
    if (rtn)
    {
        MessageBox.Show("分析成功");
    }
    else
    {
        MessageBox.Show("分析失败");
    }
    //关闭类，数据库，断开数据源连接
    _NetCls.Close();
    GDB.Close();
    _GDBServer.DisConnect();
}

using MapGIS.GeoDataBase.Net;

using MapGIS.GeoDataBase;

using MapGIS.GeoObjects.Geometry;

 

//定义变量

DataBase GDB = null;

Server _GDBServer = null;

NetAnalyse _NetAnalyse = null;

NetCls _NetCls = null;

List<LocateAnaInfo> _LocateAnaInfos = null;//定位分析结果对象

LocateAnaParam _LocateAnaParam = null;//定位分析参数设置对象

List<int> CenterIDs = null;//候选中心点ID列表

List<int> StopIDs = null;//候选目的点ID列表

//连接数据源，打开数据库         

_GDBServer = new Server();

bool rtn = _GDBServer.Connect("MapGisLocal", "", "");

if (rtn)

{

    GDB = _GDBServer.OpenGDB("DevelopDB");

    _NetCls = new NetCls(GDB);

    _NetCls.Open("net", 0);

    //变量初始化

    _NetAnalyse = new NetAnalyse();

    _LocateAnaParam = new LocateAnaParam();

    CenterIDs = new List<int>();

    StopIDs = new List<int>();

    //设置候选中心点ID列表

    Dot pos = new Dot(114.4391, 38.0771);

    int ID = _NetCls.GetNearNode(pos, 0.003);

    CenterIDs.Add(ID);

    pos = new Dot(114.4347, 38.0703);

    ID = _NetCls.GetNearNode(pos, 0.003);

    CenterIDs.Add(ID);

    //设置候选目的地ID列表

    pos = new Dot(114.4369, 38.0481);

    //获取点的ID号

    ID = _NetCls.GetNearNode(pos, 0.003);

    StopIDs.Add(ID);

    //获取点的ID号

    pos = new Dot(114.4455, 38.0513);

    ID = _NetCls.GetNearNode(pos, 0.003);

    StopIDs.Add(ID);

    //获取点的ID号

    pos = new Dot(114.4421, 30.0237);

    ID = _NetCls.GetNearNode(pos, 0.003);

    StopIDs.Add(ID);

    //设置分析参数对象

    _LocateAnaParam.CandidateCenter = CenterIDs;//设置中心候选点列表

    _LocateAnaParam.LocateCenterNum = 1;//中心点个数

    _LocateAnaParam.LimitRange = 30;//覆盖范围

    _LocateAnaParam.CandidateStop = StopIDs;//候选目的点列表

    //设置网络类

    _NetAnalyse.Cls = _NetCls;

    //定位分析

    int itemp = _NetAnalyse.LocatAnalyse(_LocateAnaParam, out _LocateAnaInfos);

    if (itemp > 0)

        MessageBox.Show("分析成功");

    }

        MessageBox.Show("分析失败");

    }

    //关闭类，数据库，断开数据源连接
