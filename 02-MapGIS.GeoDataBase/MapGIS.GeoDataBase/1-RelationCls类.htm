示例

AddSet 方法示例

C#

using MapGIS.GeoDataBase;

 

//连接数据源,打开数据库

Server srv = new Server();

bool flag = srv.Connect("MapGISLocal", "", "");

if(!flag)

{

   MessageBox.Show("连接数据源失败");

   return;

}

DataBase gdb = srv.OpenGDB("sample");

 

//打开关系类

RelationCls target = (RelationCls)(gdb.GetXClass(XClsType.RCls));

flag = target.Open("关联关系类", 0);

 

//查询

QueryDef def = new QueryDef();

def.Filter = "id=1";

RecordSet xset = target.Select(def);

 

//追加集

target.AddSet(xset);

 

//关闭数据库

target.Close();

gdb.Close();

srv.DisConnect();

Create(String,Int32,RClsInfo,Fields) 方法示例

C#

using MapGIS.GeoDataBase;

using MapGIS.GeoObjects.Att;

 

//连接数据源,打开数据库

Server srv = new Server();

bool flag = srv.Connect("MapGISLocal", "", "");

if(!flag)

{

   MessageBox.Show("连接数据源失败");

   return;

}

 

DataBase gdb = srv.OpenGDB("sample");

RelationCls target = (RelationCls)(gdb.GetXClass(XClsType.RCls));

target.Open("关联关系类", 0);

 

//设置信息

string name = "测试关系类";

int dsID = 0;  

RClsInfo info = (RClsInfo)target.ClsInfo;

Fields flds = target.Fields;

 

//创建关系类

RelationCls target1 = (RelationCls)(gdb.GetXClass(XClsType.RCls));

target1.Create(name, dsID, info, flds);

 

//关闭数据库

gdb.Close();

srv.DisConnect();

1.此示例代码需要在VS工程里的引用下面添加如下组件，分别为MapGIS.GeoDataBase.dll,MapGIS.GeoObjects.dll；

2.可以在平台安装目录下面的Program文件夹下面找到对应的组件。
