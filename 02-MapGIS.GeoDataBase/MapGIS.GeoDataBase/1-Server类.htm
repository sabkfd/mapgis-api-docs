CreateGDB 方法示例

创建地理数据库，存放到本地数据源中。

C#

using MapGIS.GeoDataBase;

 

Server svr = null;   //地理数据库服务器

svr = new Server();

bool rtn = svr.Connect("Mapgislocal", "", "");

if (rtn)

{

     //日志事件触发器

     LogEventReceiver LogER = new LogEventReceiver();

 

     //地理数据库创建信息对象

     GDBCreateParam createParam = new GDBCreateParam();

     //数据库文件信息对象

     DBFileInfo fileInfo = new DBFileInfo();

     //数据库文件信息列表对象

     List<DBFileInfo> ListDB = new List<DBFileInfo>();

     //数据库文件扩展信息对象

     FileExtendInfo extendInfo = new FileExtendInfo();

 

     //设置数据库文件扩展信息  ***这个一定要设置

     extendInfo.ExtendMode = FileExtendMode.Size; //数据增长类型

     extendInfo.ExtendSize = 1;                   //数据文件增长大小

     extendInfo.ExtendUnit = FileExtendUnit.Mbyte; //文件增长单位

     extendInfo.IsExtendable = true; //是否实现自增长

     extendInfo.MaxFileSize = 0;   //数据库最大容量

 

     //设置数据库文件信息

     fileInfo.ExtendInfo = extendInfo;  //数据库文件扩展信息

     fileInfo.FilePath = @"C:\MapGIS 10\Sample\WG.HDF"; //数据库路径

     fileInfo.InitSize = 20; //数据库初始大小

 

     //数据库文件信息列表

     ListDB.Add(fileInfo);

 

     createParam.GDBName = "WG"; //地理数据库的名称

     createParam.GDBOwner = "";  //用户名称

     createParam.OwnerPsw = "";  //用户密码

     createParam.DataFileInfos = ListDB; //数据库文件信息列表

     createParam.IndexFileInfo = fileInfo; //数据库文件信息        

           

     //创建数据库

     int i = Svr.CreateGDB(createParam, LogER);

 

     if (i > 0)

       {

          MessageBox.Show("创建成功");

       }

     else

          MessageBox.Show("创建失败");

           

     Svr.DisConnect();

}

备注

1.数据库文件扩展信息对象（FileExtendInfo 类）必须设置；

2.此示例代码需要在VS工程里的引用下面添加如下组件，分别为MapGIS.GeoDataBase.dll；

3.可以在平台安装目录下面的Program文件夹下面找到对应的组件。
