CalcRealArea方法示例

C#

using MapGIS.GeoMap.EditUtility;

using MapGIS.GeoObjects.Geometry;

using MapGIS.GeoObjects.SpatialRef;

using MapGIS.GeoDataBase;

 

//构建多点对象

Dot dot1 = new Dot(114.250, 30.63);

Dot dot2 = new Dot(114.375, 30.63);

 

Dot dot3 = new Dot(114.375, 30.68);

Dot dot4 = new Dot(114.250, 30.68);

 

Dots dots = new Dots();

dots.Append(dot1);

dots.Append(dot2);

dots.Append(dot3);

dots.Append(dot4);

 

Dots[] dotsList = new Dots[1];

dotsList[0] = dots;

 

//设置投影前后空间参照系

SRefData srefDataIn = new SRefData();

SRefData srefDataOut = new SRefData();

 

Server srv = new Server();

srv.Connect("mapgislocal","","");

DataBase db = srv.OpenGDB("平台基础示例数据");

 

//高斯大地坐标系_北京54_38带3_北

int sridIn = 53;

 

//地理坐标系(北京)_度

int sridOut = 2;

 

SpatialRefMng refMng = new SpatialRefMng(db);

srefDataIn = refMng.Get(sridIn);

srefDataOut = refMng.Get(sridOut);

 

//计算实地面积

double area = GisMath.CalcRealArea(dotsList, srefDataIn, ref srefDataOut);

MessageBox.Show(area.ToString());

 

db.Close();

srv.DisConnect();

说明：

1、mapOut一般是球面坐标系，mapIn为平面坐标系；

2、示例中的涉及的空间参照系参照，若存在于平台提供的默认空间参照系中，即可采用示例中的方法获取对应的空间参照系，若平台中没有，则通过设置SRefData对象的属性即可。

 

StringRange方法示例

C#

using MapGIS.GeoObjects.Geometry;

using MapGIS.GeoMap;

using MapGIS.GeoObjects.Info;

using MapGIS.GeoDataBase;

using MapGIS.GeoMap.EditUtility;

 

//定义变量

Server svr = null;

DataBase db = null;

IVectorCls vecCls = null;

Rect rect = null;

sbyte be =new sbyte();

 

//变量初始化

svr = new Server();

 

//连接数据源、打开数据库

svr.Connect("MapGISLocal", "", "");

db = svr.OpenGDB("DevelopDB");

 

vecCls = db.GetXClass(XClsType.ACls) as IVectorCls;

vecCls.Open("R_Dsmt1.wt", 0);

 

//获取OID=1对象的图形信息

TextAnnInfo info = vecCls.GetInfo(1) as TextAnnInfo;

TextAnno textAnno = vecCls.GetGeometry(1) as TextAnno;

 

//控制点坐标

Dot dot = textAnno.AnchorDot;

//字符串内容

string str = textAnno.Text;

 

//获取字符串的范围

int rtn = GisMath.StringRange(dot.X, dot.Y, info.Height, info.Width, str, info.Ifnt.ToString(), info.Angle, info.Space, be, ref rect);

 

if (rtn > 0)

    MessageBox.Show("取闭包矩形成功");

else

    MessageBox.Show("取闭包矩形失败");

 

return;
