示例

AddRenderElement 方法示例

C#

 

using MapGIS.Scene3D;

using MapGIS.GeoMap;

using MapGIS.GeoDataBase;

using MapGIS.GeoObjects.Geometry;

  Server svr = new Server();

  SceneControl sceneCtrl = new SceneControl();

  //设置场景显示模式为平面

  sceneCtrl.Mode = SceneMode.LOCAL;

  //构建场景对象

  Scene scene = new Scene();

  //设置场景名称

  scene.Name = "新场景1";

  sceneCtrl.Append(scene);

  //连接数据源

  svr.Connect("mapgislocal", "", "");

  SFeatureCls sfcls = new SFeatureCls();

  //根据url打开简单要素类

  bool rtn = sfcls.Open("gdbp://MapGisLocal/示例数据/ds/三维示例/sfcls/景观_模型");

  if (rtn)

  {

      //创建模型图层对象

      ModelLayer modelLayer = new ModelLayer();

      //关联数据

      modelLayer.AttachData(sfcls);

      Scene scene = this.sceneCtrl.GetScene(0);

      //添加模型图层到场景中

      scene.Append(modelLayer);

      //更新场景视图

      this.sceneCtrl.Refresh();

      //复位显示

      this.sceneCtrl.Reset();

      //获取粒子元素，烟花特效

      ParticleElement particleElement = ParticleElement.GetUserParticleFromTemplate(DefParticleType.FireWork);

      if (particleElement != null)

      {

          //取场景中点，在中点处添加特效

          Rect3D rect3d = new Rect3D();

          scene.GetExtent(ref rect3d);

          double x = (rect3d.XMax + rect3d.XMin) / 2;

          double y = (rect3d.YMax + rect3d.YMin) / 2;

          double z = (rect3d.ZMax + rect3d.ZMin) / 2;

          Dot3D dot3d = new Dot3D(x, y, z);

          //特效位置设置

          particleElement.Position = dot3d;

          //获取一个粒子元素里面第1个粒子效果

          ParticleEffect particleEffect = particleElement.GetParticleEffect(0);

          //获取粒子发射器

          Emitter emitter = particleEffect.GetEmitter(0);

          //设置粒子发射器相对场景节点位置

          Dot3D pos = new Dot3D(0, 0, 16);

          emitter.Offset = pos;

          //每秒发射的粒子数

          emitter.EmissionRate = 5000;

          //粒子发射器的最大偏移角度

          emitter.Angle = 180;

          //添加到场景            

          scene.AddRenderElement(particleElement);

          //获取离子特效个数

          int num = scene.GetRenderElementNum(0);

          //更新场景,场景中屏幕元素变化之后更新场景

          sceneCtrl.UpdateScene(scene, 0, num - 1, LayerOperType.SetLayerAppend);

      }

备注：1.通过本方法，可以获取默认的粒子特效，该示例为在场景的中心添加烟花特效；2.此示例代码需要在VS工程里的引用下面添加如下组件，分别为MapGIS.GeoDataBase.dll,MapGIS.GeoObjects.dll，MapGIS.GeoMap.dll。

 

Append 方法示例

C#

 

using MapGIS.GeoMap;

using MapGIS.Scene3D;

 

Scene scene = null;

//二维图层

Vector3DLayer layer = null;

 

//初始化矢量图层对象

layer = new Vector3DLayer();

//设置矢量图层数据Url

layer.URL = "gdbp://MapGisLocal/HNSFDLG01/ds/25w/sfcls/XZQ_XZ_25w_polygon";

//设置图层显示名称

layer.Name = "hnerweiLayer";

//设置图层状态

layer.State = LayerState.Visible;

 

//初始化场景对象

scene = new Scene();

//添加图层到场景对象中

scene.Append(layer);

 

//场景添加到场景控件用于显示

this.sceneCtrl.Append(scene);

RemoveRenderElement 方法示例

C#

 

using MapGIS.Scene3D;

using MapGIS.GeoMap;

using MapGIS.GeoDataBase;

                         

 

Document doc = new Document();

int t = doc.Open("E:\示例文档.mapx");

if (t > 0)

{

  //获取当前场景

  Scene scene = this.sceneCtrl.GetScene(0);

 //获取场景中粒子的个数，0为粒子效果

  int num = scene.GetRenderElementNum(0);

//移除粒子特效，第一个参数是粒子效果，第二个参数是要删除元素的索引序号

  bool t = scene.RemoveRenderElement(0, num - 1);

 //更新场景，场景中屏幕元素变化之后调用，这里是移除粒子特效，最后一个参数选择SetLayerDelete方式

  bool r = this.sceneCtrl.UpdateScene(scene, 0, num - 1, LayerOperType.SetLayerDelete);

}

备注：此示例代码需要在VS工程里的引用下面添加如下组件，分别为MapGIS.GeoDataBase.dll,MapGIS.GeoMap.dll。
