示例

CreateInstance 方法示例

该示例为在三维场景中添加服务图层（天地图），成功运行本示例，需在Form中添加SceneControl控件。

C#

using MapGIS.Scene3D;

using MapGIS.GeoMap;

                

Scene scene = new Scene();

//获取场景数

int count = this.sceneCtrl.GetSceneNum();

if (count > 0)

{

    //获取场景

    scene = this.sceneCtrl.GetScene(0);

}

else

{

    //添加场景

    this.sceneCtrl.Append(scene);

}

//地图服务类

MapServer mapserver = null;

//地图服务类型信息

MapServerInfo[] serverInfos = null;

serverInfos = MapServer.GetServerInfos();

//地图服务类型信息

MapServerInfo serverInfo=null ;

//本地瓦片服务

string type = "Tianditu";

foreach (MapServerInfo info in serverInfos)

{

    //判断服务类型为当前类型时

    if (info.ServerType == type || info.ServerType.Equals(type))

    {

        //MapServerInfo类型，当前服务信息

        serverInfo = info;

        break;

    }

}

//根据服务类型信息创建地图服务

mapserver = MapServer.CreateInstance(serverInfo);

string URL = @"http://t0.tianditu.com/vec_c/wmts";

//本地瓦片路径

mapserver.URL = URL;

 

//根据URL获取服务图层名

string name = URL;

if (!string.IsNullOrEmpty(name))

{

    int index = name.IndexOf(":");

    if (index >= 0)

        name = name.Substring(index + 1);

    index = name.IndexOf("?");

    if (index >= 0)

        name = name.Substring(0, index);

    name = name.Replace("\\", "/");

    index = name.LastIndexOfAny(new char[] { '\\', '/' });

    if (index >= 0)

        name = name.Substring(index + 1);

}

if (string.IsNullOrEmpty(name))

    name = "ServerLayer";

       

//创建三维服务图层

Server3DLayer server3Dlayer = new Server3DLayer();

//三维服务图层的地图服务

server3Dlayer.MapServer = mapserver;  

//连接数据

server3Dlayer.ConnectData();

server3Dlayer.Name = name;

//添加地形图层到场景中

int k = scene.Append(server3Dlayer);

//服务场景显示控件

this.sceneCtrl.Reset();

mapserver.Dispose();
