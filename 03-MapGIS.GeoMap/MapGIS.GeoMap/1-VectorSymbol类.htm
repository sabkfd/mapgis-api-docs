示例

GetItem 方法示例

本示例为获取MapGIS 10系统库中的某一个点状子图符号的具体构造图元。可在此基础上，编辑子图符号。

C#

using MapGIS.GeoMap;

using MapGIS.GeoObjects;

 

//系统库管理对象

SystemLibrarys sysLibs = SystemLibrarys.GetSystemLibrarys();

int sysLibsCount = sysLibs.Count;

               

for (int i = 0; i < sysLibsCount; i++)

{

    SystemLibrary sysLib = sysLibs.GetSystemLibrary(i);

    if (sysLib.Name == "MapGIS 10")

    {                                  

        SymbolLibrary s = sysLib.GetSymbolLibrary();

        Symbols symbols=new Symbols ();

        //选择符号库

        if (s.SelectSymbols(SymbolGeomType.GeomPnt, ref symbols))

        {

            //获取点符号

            VectorSymbol symbol = symbols.GetSymbol(SymbolGeomType.GeomPnt, 7, 7) as VectorSymbol;                    

            string name = symbol.Name;

          

          for(int j=0;j<=symbol.Count;j++)

          {

              //获取符号的各个子项

              SymbolItem symitem = symbol.GetItem(j);                        

              switch (symitem .Type )

              {

                  case SymbolItemType .Polygon :

                      PolygonSymbolItem polysymbol = symitem as PolygonSymbolItem;

                     SDot[] polydot= polysymbol.GetDots();

                      break;

                  case SymbolItemType.PolyLine :

                      PolyLinSymbolItem linesymbol = symitem as PolyLinSymbolItem;

                      SDot[] linedot = linesymbol.GetDots();

                      break;

                  case SymbolItemType.Arc :

                      ArcSymbolItem acrsymbol = symitem as ArcSymbolItem;

                      SDot arcdot = acrsymbol.Center;

                      break; 

             }                      

          }                      

        }

        break;

    }

}
