RangeLabel类

分段标注图(支持单表达式分段)

备注：分段动态注记与分段专题图类似，只是其用注记来展现图形业务属性，例如通过注记大小、颜色等等样式参数来分类展现。

示例

Append 方法示例

C#

using MapGIS.GeoMap;

using MapGIS.GeoObjects.Geometry;

using MapGIS.GISControl;

using MapGIS.GeoObjects.Info;

using MapGIS.GeoDataBase;

using MapGIS.GeoObjects;

 

//定义变量

Server srv = null;

DataBase db = null;

VectorLayer vectorLayer = null;

IVectorCls VectorCls = null;

Map map = null;

 

//连接数据源，打开数据库

srv = new Server();

srv.Connect("MapGislocal", "", "");

db = srv.OpenGDB("DevelopDB");

 

//打开数据

VectorCls = db.GetXClass(XClsType.SFCls) as IVectorCls;

VectorCls.Open("平若县行政略图.WP", 0);

//关联数据到图层

vectorLayer = new VectorLayer(VectorLayerType.SFclsLayer);

vectorLayer.AttachData(VectorCls);

 

//第一分段的注记参数信息

TextAnnInfo annInfo = new TextAnnInfo();

annInfo.Bold = true;

annInfo.Color = 6;

annInfo.Height = 20;

annInfo.Width = 20;

 

//第二分段的注记参数信息

TextAnnInfo annInfo1 = new TextAnnInfo();

annInfo1.Bold = true;

annInfo1.Color = 5;

annInfo1.Height = 30;

annInfo1.Width = 30;

 

//未参与分段的注记参数信息

TextAnnInfo defaultAnnInfo = new TextAnnInfo();

defaultAnnInfo.Bold = false;

defaultAnnInfo.Color = 9;

defaultAnnInfo.Height = 30;

defaultAnnInfo.Width = 30;

 

//动态注记信息

LabelInfo labelInfo = null;

labelInfo = new LabelInfo();

labelInfo.AnnInfo = defaultAnnInfo;

labelInfo.Caption = "未参与分类";

labelInfo.IsVisible = true;

 

//分段动态注记

RangeLabel rangeLbl = null;

rangeLbl = new RangeLabel();

 

RangeThemeInfo themeInfo = null;

RangeThemeInfo themeInfo1 = null;

 

themeInfo = new RangeThemeInfo();

themeInfo1 = new RangeThemeInfo();

 

//构建分段专题图的分段信息

 

//第一分段

themeInfo.IsVisible = true;

themeInfo.StartValue = "16";

themeInfo.EndValue = "39";

themeInfo.Caption = "zondy";

themeInfo.GeoInfo = annInfo;

 

//第二分段

themeInfo1.IsVisible = true;

themeInfo1.StartValue = "39";

themeInfo1.EndValue = "62";

themeInfo1.Caption = "zondy1";

themeInfo1.GeoInfo = annInfo1;

 

rangeLbl.Append(themeInfo);

rangeLbl.Append(themeInfo1);

rangeLbl.SetDefaultInfo(labelInfo);

//两个属性均须给

rangeLbl.Expression = "总产";

rangeLbl.LabelExpression = "总产";

rangeLbl.Visible = true;

 

//添加到图层并显示

vectorLayer.Label = rangeLbl;

map = new Map();

map.Append(vectorLayer);

this.mapControl1.ActiveMap = map;

this.mapControl1.Restore();

 

//关闭类，数据库，断开数据源连接

VectorCls.Close();

db.Close();

srv.DisConnect();

备注

1.此示例代码需要在VS工程里的引用下面添加如下组件，分别为MapGIS.GISControl.dll,MapGIS.GeoMap.dll,MapGIS.GeoObjects.dll,MapGIS.GeoDataBase.dll；

2.可以在平台安装目录下面的Program文件夹下面找到对应的组件。
