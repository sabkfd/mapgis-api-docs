示例

ChartTheme示例

C#

 
using MapGIS.GeoMap;
using MapGIS.GeoObjects.Geometry;
using MapGIS.GISControl;
using MapGIS.GeoObjects.Info;
using MapGIS.GeoDataBase;
using MapGIS.GeoObjects;
 
//定义变量
Server srv = null;
DataBase db = null;
VectorLayer vectorLayer = null;
IVectorCls VectorCls = null;
Map map = null;
 
ChartTheme chartTheme = null;
ChartThemeInfo chartThemeInfo = null;
ChartThemeInfo chartThemeInfo1 = null;
 
//连接数据源，打开数据库
srv = new Server();
srv.Connect("MapGislocal", "", "");
db = srv.OpenGDB("DevelopDB");
 
//打开简单要素类
VectorCls = db.GetXClass(XClsType.SFCls) as IVectorCls;
VectorCls.Open("平若县行政略图.WP", 0);
//关联数据到图层
vectorLayer = new VectorLayer(VectorLayerType.SFclsLayer);
vectorLayer.AttachData(VectorCls);
 
 
chartTheme = new ChartTheme();
 
//构建统计图图形信息
chartThemeInfo = new ChartThemeInfo();
chartThemeInfo.Caption = "zondy";
chartThemeInfo.Expression = "水稻";
chartThemeInfo.IsVisible = true;
 
RegInfo rInfo = new RegInfo();
rInfo.FillClr = 3;
rInfo.Ovprnt = true;
chartThemeInfo.GeoInfo = rInfo;
 
chartThemeInfo1 = new ChartThemeInfo();
chartThemeInfo1.Caption = "zondy1";
chartThemeInfo1.Expression = "小麦";
chartThemeInfo1.IsVisible = true;
 
RegInfo rInfo1 = new RegInfo();
rInfo1.FillClr = 4;
rInfo1.Ovprnt = true;
chartThemeInfo1.GeoInfo = rInfo1;
 
//设置统计专题图类型
chartTheme.ChartType = ChartType.Pie;
//统计图名称
chartTheme.Name = "平若县行政略图.WP";
 
//配置统计图样式等信息
ChartThemeRepresentInfo chartThemeRepresentInfo = null;
chartThemeRepresentInfo = new ChartThemeRepresentInfo();
//小数位数
chartThemeRepresentInfo.DigitLabel = 1;
//实际值
chartThemeRepresentInfo.FormatLabel = ChartLabelFormat.Value;
//各饼图采用相同大小
chartThemeRepresentInfo.PieSizeFixFlag = 1;
//饼图最小半径
chartThemeRepresentInfo.MinRadius = 5;
//设计统计图样式
chartTheme.SetRepresentInfo(chartThemeRepresentInfo);
 
//添加统计题图的字段表达式信息
chartTheme.Append(chartThemeInfo);
chartTheme.Append(chartThemeInfo1);
//添加新创建的专题图到图层
vectorLayer.Themes.Append(chartTheme);
 
//构建地图
map = new Map();
//添加图层
map.Append(vectorLayer);
 
//激活地图
this.mapControl1.ActiveMap = map;
this.mapControl1.Restore();
 
//关闭类，数据库，断开数据源连接
VectorCls.Close();
db.Close();
srv.DisConnect();

备注：统计专题图可以在一幅地图上表现多个属性以及不同属性间的关系，也可以用直方图、饼图、折线图、散点图、柱状图等形式来表示要素的结构。比如有一个县的粮食产量数据，可以对其不同粮食作物的产量进行统计，用直方图或柱状图来显示各地区各种作物的产量，或者用饼图表示某个区各作物在粮食总产量中所占的比例，以此对各个区各种粮食产量做直观的分析。注：统计专题图只支持数值型的数据。
