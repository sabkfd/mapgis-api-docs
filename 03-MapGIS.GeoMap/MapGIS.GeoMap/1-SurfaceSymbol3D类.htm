示例

GetBitMap 方法示例

此处示例主要是介绍获取获取二维浏览位图.

C#

using MapGIS.GeoMap;

 

/// <summary>

        /// 获取符号的图片

        /// </summary>

        /// <param name="picBox">图片控件</param>

        /// <param name="sysLib">系统库对象</param>

        /// <param name="id">符号ID</param>

        /// <returns>符号名称</returns>

        public Symbol3D GetSymbolImage(PictureBox picBox,SystemLibrary sysLib, string id)

        {

            //三维符号库

            Symbol3DLibrary sym3DLib = sysLib.GetSymbol3DLibrary();

            //获取指定ID的符号

            Symbol3D sym3D = null;

            //图片二进制流

            byte[] imageByte = null;

            //符号名称

            string symbolName = "";

 

            //获取指定ID的符号

            sym3DLib.GetSymbol(int.Parse(id), 0, out sym3D);                      

            Symbol3D symbol = null;

            switch (sym3D.GetType())

            {

                case SymbolType.Pnt3D: symbol = (PointSymbol3D)sym3D; break;

 

                case SymbolType.Lin3D: symbol = (LineSymbol3D)sym3D; break;

 

                case SymbolType.Sur3D: symbol = (SurfaceSymbol3D)sym3D; break;

                    

                default: break;

            }

 

            //获取图片和符号名称

            imageByte = symbol.GetBitMap();

            symbolName = symbol.GetName();

 

            if (imageByte != null)

            {

                MemoryStream stream = new MemoryStream(imageByte);

                Image image = Image.FromStream(stream);

                picBox.Image = image;

                picBox.SizeMode = PictureBoxSizeMode.StretchImage;

            }

 

            return sym3D;

        }

 

GetItem 方法示例

此处代码主要是介绍了根据索引获取符号项

C#

using MapGIS.GeoMap;

 

  private void comboBox1_SelectedIndexChanged(object sender, EventArgs e)

        {

            this.sceneCtrl.RemoveAll();

            if (symbol.GeomType == SymbolGeomType.GeomSur3D)

            {

                //获取面符号第一项的颜色

                SurSymbolItem surItem = null;

                surfaceSymbol.GetItem(0, ref surItem);

                surItem.SetFillColor(int.Parse(this.comboBox1.Text), true);

                surfaceSymbol.UpdateItem(0, surItem);

                //更新符号颜色

                sysLib.GetSymbol3DLibrary().UpdateSymbol(surfaceSymbol);

                this.sceneCtrl.Append(sysLib.Guid, surfaceSymbol, 0);

            }

            else if (symbol.GeomType == SymbolGeomType.GeomLin)

            {

                //获取符号颜色

                lineSymbol.SetFillColor(int.Parse(this.comboBox1.Text));

                //更新符号颜色

                sysLib.GetSymbol3DLibrary().UpdateSymbol(lineSymbol);

                this.sceneCtrl.Append(sysLib.Guid, lineSymbol, lineSymbol.SymbolNo);

            }

            else

            {

                //获取符号颜色

                potSymbol.SetFillColor(int.Parse(this.comboBox1.Text));              

                //更新符号颜色

                sysLib.GetSymbol3DLibrary().UpdateSymbol(potSymbol);

                this.sceneCtrl.Append(sysLib.Guid, potSymbol, 0);

            }

            return;

        }

 

SetBitMap 方法示例

C#

using MapGIS.GeoMap;

 

SystemLibrarys sysLibs = SystemLibrarys.GetSystemLibrarys();

SystemLibrary sysLib = null;

Symbol3DLibrary sys3DLib = null;

 

if (sysLibs != null)

{

    Guid guid = new Guid();

    sysLib = sysLibs.GetSystemLibrary(guid);

    if (sysLib != null)

    {

        sys3DLib = sysLib.GetSymbol3DLibrary();

    }

}

Resource3DLibrary m_Resource3DLib = sysLib.GetResource3DLibrary();

Resource3D res3D = new Resource3D();

int j = m_Resource3DLib.FormResource(@"C:\Users\wujiabei\Desktop\test.tif", null, 0, ref res3D);

if (j < 0)

{

    MessageBox.Show("获取资源失败！");

    return;

}

m_Resource3DLib.AddResource(res3D); //添加资源

int resID = res3D.GetNo(); //获取图片资源ID

m_Resource3DLib.Save();

 

//表面符号类

SurSymbolItem SurItem = new SurSymbolItem();

Symbol3D sym3D = null;

if (resID > 0)

{

    SurfaceSymbol3D surSym3D = new SurfaceSymbol3D();

    //设置符号名称

    surSym3D.SetName("DemoSym3D52");

 

    //获取图片高、宽信息

    Image image = null;

    image = Image.FromFile(@"C:\Users\wujiabei\Desktop\test.tif");

    int newWidth = image.Width;

    int newHeight = image.Height;

 

    byte[] buff = null;

    MemoryStream stream = new MemoryStream();

    Image img = image.GetThumbnailImage(newWidth, newHeight, null, IntPtr.Zero);

    img.Save(stream, image.RawFormat);

    buff = stream.GetBuffer();

 

    //构造面符号项类

    SurSymbolItem surItem = new SurSymbolItem();

    //填充色

    surItem.SetFillColor(5, true);

    //设置透明度

    surItem.SetTransparentClr(0, true);

    surItem.SetTexType(1, true);

    surItem.SetTrimType(1, true);

    surItem.SetTexPic(resID, true);

    //设置纹理缩放

    surItem.SetZoom(1, 1, true);

    //设置纹理单元高宽

    surItem.SetUnitSize((float)(newWidth / 100.0), (float)(newHeight / 100.0), true);

    //添加符号项

    surSym3D.AddItem(surItem);

    sym3D = surSym3D;

    if (sym3D != null)

    {

        //设置二维浏览位图

        sym3D.SetBitMap(buff);

 

        //添加符号到最后,返回该类型的编号

        int i = sys3DLib.AddSymbol(sym3D);

        if (i > 0)

        {

            MessageBox.Show("添加成功！");

        }

        else

        {

            MessageBox.Show("添加失败！");

        }

    }

}
