示例

Append 方法示例

C#

 
using MapGIS.GeoMap;
using MapGIS.GISControl;
using MapGIS.GeoObjects.Geometry;
using MapGIS.GeoDataBase;
using MapGIS.GeoObjects.Info;
using MapGIS.GeoObjects;
 
//取版面视图控件的版面对象
Layout lt = this.layoutControl1.ActiveLayout;
if (lt != null)
{
    int with = 100;
    int height = 50;
    //1、选择要添加的图片文
        using (Image img = Image.FromFile(@"E:\7-图片素材\图标\61666.png"))
        {
            if (img != null)
            {
                with = img.Width;
                height = img.Height;
                img.Dispose();
            }
        }
    //设置图片元素显示的位置屏幕坐标（200，200）
    double lx = 0;
    double ly = 0;
    //版面视图控件须先拖到form窗体中，获取对应的逻辑坐标
    this.layoutControl1.Transformation.WpToLp(200, 200, ref lx, ref ly);
    Rect rt = new Rect(lx, ly, lx + with, ly + height);
    //设置版面控件的历史操作记录
    this.layoutControl1.EditHistory.Start(this.ToString());
    this.layoutControl1.EditHistory.AppendDevEnv(lt.Handle);
    //2、创建图片元素，设置相关参数
    PictureElement pe = new PictureElement();
    pe.IsSavePictureInDocument = true;
    pe.PictureFilePath =  @"E:\7-图片素材\图标\61666.png";
    pe.IsFixedAspectRatio = true;
    pe.SetGeometry(rt);
    pe.SetRange(rt);
    pe.SetViewRange(rt);
    //图片元素添加到版面对象中
    lt.Append(pe);
    //设置版面操作原子
    LayoutAtomOperData laod = new LayoutAtomOperData();
    //设置当前为图片操作
    laod.OperElementType = OperationElementType.Picture;
    laod.OperType = OperationType.Append;
    //设置当前操作
    laod.SetOperData(pe, true);
    //存到历史记录
    this.layoutControl1.EditHistory.Append(laod);
    this.layoutControl1.EditHistory.End();
    //刷新版面元素
    layoutControl1.RefreshElement(pe, false);
}

备注：可添加文本、图片、几何、指北针等版面元素。此示例代码需要在VS工程里的引用下面添加如下组件，分别为MapGIS.GeoMap.dll,MapGIS.GeoObjects.dll, MapGIS.GISControl.dll，MapGIS.GeoDataBase.dll。

OutputToEPSFile 方法示例

C#

using MapGIS.GeoMap;

using MapGIS.GISControl;

 

//文档对象

Document doc = null;

doc = new Document();

//打开文档

int rtn = doc.Open(@"E:\18-Data\data-ol\Sample\sample_wuhan.mapx");

//在form设计页面拖入版面视图控件layoutControl，设为版面视图

this.layoutControl1.ActiveLayout = doc.GetLayout();

this.layoutControl1.Refresh();

//取版面视图控件当前激活的版面类

Layout lt = layoutControl1.ActiveLayout;

if (lt != null)

{

    //创建保存文件对话框，设置文件格式

    SaveFileDialog sfd = new SaveFileDialog();

    sfd.Filter = "PS文件(*.ps)|*.ps|EPS文件(*.eps)|*.eps";

    if (sfd.ShowDialog() == DialogResult.OK)

    {

        //设置文件的高宽、路径

        double with = 0;

        double height = 0;

        lt.GetLayoutSize(ref with, ref height);

        string fileName = sfd.FileName;

        lt.OutputToEPSFile(fileName, (int)with, (int)height, false);

    }

}
