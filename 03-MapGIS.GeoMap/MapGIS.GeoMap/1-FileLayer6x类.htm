示例

FileLayer6x 类示例

C#

using MapGIS.GeoDataBase;

using MapGIS.GeoMap;

 

Document doc = new Document();

int i = doc.Open(@"F:\6X数据地图文档.mapx");

if (i <= 0)

{

    MessageBox.Show("打开失败！");

    return;

}

Map map = doc.GetMaps().GetMap(0);

if (map != null)

{

    List<MapLayer> mapLayerlist = map.GetEditLayer(EditLayerType.Line, SelectLayerControl.Active);

    int layerCount = map.LayerCount;

 

    FileLayer6x FileLayer6xLine = new FileLayer6x();

    if (layerCount < 0) return;

    FileLayer6xLine = map.get_Layer(0) as FileLayer6x;

 

    SFeatureCls sfclsLine = new SFeatureCls();

    if (FileLayer6xLine == null) return;

    //2004等值线

    sfclsLine = FileLayer6xLine.get_Item(0).GetData() as SFeatureCls;

    //创建点图层

    SFeatureCls sfclsPoint = new SFeatureCls();

    if (!sfclsPoint.Open(@"gdbp://MapGisLocal/DMtest/sfcls/水位点"))

    {

        MessageBox.Show("结果图层打开失败！");

        sfclsLine.Close();

        sfclsPoint.Close();

        return;

    }

    createPoint(sfclsLine, sfclsPoint);

    sfclsLine.Close();

    sfclsPoint.Close();                        

}

SaveOrGiveUpEdit 方法示例

C#

using MapGIS.GeoDataBase;

using MapGIS.GeoObjects.Geometry;

using MapGIS.GeoObjects;

using MapGIS.GeoMap;

 

SFeatureCls sfcls = new SFeatureCls();

if (sfcls.Create("file:///" + @"F:\测试区图层1.wp", GeomType.Reg) > 0)

{

    //构造点

    Dots dots = new Dots();

    Dot dot = new Dot(0, 0);

    Dot dot1 = new Dot(100, 0);

    Dot dot2 = new Dot(0, 100);

    Dot dot3 = new Dot(100, 100);

 

    dots.Append(dot);

    dots.Append(dot1);

    dots.Append(dot3);

    dots.Append(dot2);

    dots.Append(dot);

 

    //构造区

    GeoVarLine line = new GeoVarLine(dots);

    GeoLines lines=new GeoLines ();

    lines .Append (line);

    GeoPolygon plygon=new GeoPolygon ();

    plygon .Append (lines);

 

    //添加要素

    if (sfcls.Append(plygon, null, null) > 0)

    {

        VectorLayer vecLayer = new VectorLayer(VectorLayerType.SFclsLayer);

        if (vecLayer.AttachData(sfcls))

        {

            //创建6X图层

            FileLayer6x layer = new FileLayer6x();

            //设置6X图层路径

            layer.URL = vecLayer.URL;

            layer.ConnectData();

 

            //添加点简单要素类到6X图层中

            if (layer.Append(vecLayer) <= 0) return;

            //是否保存数据

            if (!layer.SaveOrGiveUpEdit(true)) MessageBox.Show("数据保存失败！","提示");

           

            //6Ｘ数据显示

            Document doc = this.mapTree.Document;

            Map map = new Map();

            if (map.Append(layer) <= 0) return;

            doc.GetMaps().Append(map);

 

            this.mapCtrl.ActiveMap = map;

            this.mapCtrl.Restore();

        }

    }

}

else

{

    MessageBox.Show("创建图层失败");

}
