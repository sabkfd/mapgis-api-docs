示例

Append 方法示例

C#

using MapGIS.GeoMap;

using MapGIS.GISControl;

using MapGIS.GeoDataBase;

 

//定义变量

Server srv = null;

DataBase db = null;

VectorLayer vectorLayer = null;

IVectorCls VectorCls = null;

Map map = null;

//定义一个MapControl控件对象

MapControl mapControl1 = new MapControl();

mapControl1.Dock = DockStyle.Fill;

this.panel1.Controls.Add(mapControl1); //把MapControl控件承载在Panel控件上面

 

//连接数据源，打开数据库

srv = new Server();

srv.Connect("MapGislocal", "", "");

db = srv.OpenGDB("DevelopDB");

if(db == null)

{

   MessageBox.Show("打开数据库失败");

   return;

}

//打开简单要素类

VectorCls = db.GetXClass(XClsType.SFCls) as IVectorCls;

VectorCls.Open("Ku6_1s.wl", 0);

 

//关联数据到图层

vectorLayer = new VectorLayer(VectorLayerType.SFclsLayer);

vectorLayer.AttachData(VectorCls);

 

//构建地图

map = new Map();

//添加图层

map.Append(vectorLayer);

//激活地图

mapControl1.ActiveMap = map;

mapControl1.Restore();

 

//关闭类，数据库，断开数据源连接

VectorCls.Close();

db.Close();

srv.DisConnect();

DragIn 方法示例

说明：将图层vectorLayer1拖动到图层vectorLayer的前面。

C#

 

using MapGIS.GeoMap;

using MapGIS.GISControl;

using MapGIS.GeoDataBase;

 

//定义变量

Server srv = null;

DataBase db = null;

VectorLayer vectorLayer = null;

VectorLayer vectorLayer1 = null;

IVectorCls VectorCls = null;

IVectorCls VectorCls1 = null;

Map map = null;

 

//定义一个MapControl控件对象

MapControl mapControl1 = new MapControl();

mapControl1.Dock = DockStyle.Fill;

this.panel1.Controls.Add(mapControl1); //把MapControl控件承载在Panel控件上面

 

//连接数据源，打开数据库

srv = new Server();

srv.Connect("MapGislocal", "", "");

db = srv.OpenGDB("DevelopDB");

if(db == null)

{

   MessageBox.Show("打开数据库失败");

   return;

}

//打开简单要素类

VectorCls = db.GetXClass(XClsType.SFCls) as IVectorCls;

VectorCls.Open("Ku6_1s.wl", 0);

//关联数据到图层

vectorLayer = new VectorLayer(VectorLayerType.SFclsLayer);

vectorLayer.AttachData(VectorCls);

 

//打开简单要素类

VectorCls1 = db.GetXClass(XClsType.SFCls) as IVectorCls;

VectorCls1.Open("BOU1_4M.WP", 0);

//关联数据到图层

vectorLayer1 = new VectorLayer(VectorLayerType.SFclsLayer);

vectorLayer1.AttachData(VectorCls1);

 

//构建地图

map = new Map();

//添加图层

map.Append(vectorLayer);

//拖入图层到图层列表中第1个位置（图层顺序索引为0）

map.DragIn(0, vectorLayer1);

 

//激活地图

mapControl1.ActiveMap = map;

mapControl1.Restore();

 

//关闭类，数据库，断开数据源连接

VectorCls.Close();

VectorCls1.Close();

db.Close();

srv.DisConnect();

GetBookMarks 方法示例

C#

 

using MapGIS.GeoMap;

using MapGIS.GeoObjects.Geometry;

using MapGIS.GISControl;

 

Document doc = null;

Maps maps = null;

Map map = null;

 

//定义一个MapControl控件对象

MapControl mapControl1 = new MapControl();

mapControl1.Dock = DockStyle.Fill;

this.panel1.Controls.Add(mapControl1); //把MapControl控件承载在Panel控件上面

 

//打开地图文档

doc = new Document();

doc.Open(@"F:\10\MapGIS10.2\Sample\wuhan.xml");

maps = doc.GetMaps();

map = maps.GetMap(0);

 

//获取书签集

BookMarks bookMarks = map.GetBookMarks();

 

BookMark bookMark = new BookMark();

//设置当前地图显示范围

bookMark.SetLocation(map.GetViewRange());

bookMark.Name = "书签0";

 

//添加书签到书签集

bookMarks.Append(bookMark);

 

//显示地图

mapControl1.ActiveMap = map;

mapControl1.Restore();

//保存地图文档

doc.Save();

OutputToImageFile 方法示例

C#

 
using MapGIS.GeoMap;
using MapGIS.GeoObjects.Geometry;
using MapGIS.GISControl;
 
//变量定义
MapControl mapControl1;
Document doc = null;
Maps maps = null;
Map map = null;
 
//加载地图显示控件
mapControl1 = new MapControl();
mapControl1.Dock = DockStyle.Fill;
mapControl1.BackColor = System.Drawing.Color.White;
this.panel1.Controls.Add(mapControl1);
 
//打开地图文档
doc = new Document();
int irtn = doc.Open(@"F:\10\MapGIS10.2\Sample\wuhan.xml");
if (irtn <= 0)
{
   MessageBox.Show("打开地图文档失败");
   return;
}
maps = doc.GetMaps();
map = maps.GetMap(0);
 
//显示地图
mapControl1.ActiveMap = map;
mapControl1.Restore();
 
//输出为JPG图片
map.OutputToImageFile(@"F:\wuhan.jpg", mapControl1.ViewWidth, mapControl1.ViewHeight, ImgType.JPG, true, 0, false);

备注：

（1）参数fileName与imageType保持一致，例如imageType设置为ImgType.JPG，那么fileName应该给“F:\wuhan.jpg”，而最好别给例如“F:\wuhan.png”或者“F:\wuhan”，当然，OutputToImageFile方法会按照imageType指定的图片格式调用相关方法生成对应的图片；

（2）antialiasing，true，则采用反走样模式，flase则是走样模式，两种技术不同，反走样模式相对分辨率高些。
