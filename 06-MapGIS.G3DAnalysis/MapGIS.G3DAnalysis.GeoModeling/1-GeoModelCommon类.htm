static CreateClosedSurByReg(

   IGeometry3D geom,

   GeoPolygon range,

   float gGeoHeight,

   float fHeightOffSet,

   RasterDataSet dataSet,

   double terrainScale

)

int

成功返回值大于0

geom: 用来存放创建的封闭面或者几何体

range: 创建的封闭面或几何体的范围

static CreateClosedSursByReg(

   SFeatureCls srcSFcls,

   SFeatureCls dstSFcls,

   string strHeight,

   GeomInfo surInfo,

   bool isUseSrcCol,

   TransGlobeModel isGlobeMode,

   Dot3D offset,

   short attCopyMode,

   string strHeightOffSet,

   RasterDataSet dataSet,

   double terrainScale,

   MPDNativeInterface hCallBackProc

)

int

成功建模的要素个数

根据区要素类批量创建封闭的面，如建筑物

srcSFcls: 源要素类，为区类型或者体类型

dstSFcls: 已打开的三维要素类，用于存储三维要素到数据库，不能为空

strHeight: 建筑物高度的字段名或者是查询语句，如字段名*2

surInfo: 生成的封闭面的符号信息，不能为NULL

isUseSrcCol: 是否使用源要素类的颜色, true:使用； false：不使用

isGlobeMode: 是否生成球面模式的数据

offset: 生成椭球面数据是的偏移值 , 图层范围最小值xy z = 0

attCopyMode: 要素类的转换模式，0:不拷贝源类的属性结构和属性，属性值由用户自己设定;1:只拷贝源类的属性结构，属性值由用户设定; 2:拷贝源类的属性结构和属性，用户无需自己设定值

strHeightOffSet: 封闭面(几何体)的地面相对于地形的高度的字段名或者是查询语句，当地形高度为NULL时该值为地形高度

dataSet: 地形数据集,用来获取地形对应点的高程值,为空则地形基准为0

terrainScale: 地形高程缩放比

hCallBackProc: 回调参数，获取成功转换的个数

        {

            SFeatureCls sfcls = new SFeatureCls(db);

            sfcls.Create("管道0321", GeomType.Surface, 0, 0, null);

 

            //多面对象

            GeoMultiSurface multiSurface = new GeoMultiSurface();

          

            //二维折线对象

            GeoVarLine varLine = new GeoVarLine();

 

            //设置折线

            Dot dot1 = new Dot(100, 100);

            Dot dot2 = new Dot(1000, 1000);

            varLine.Append(dot1);

            varLine.Append(dot2);

 

            //二维折线对象

            GeoVarLine varLine3D = new GeoVarLine();

 

            //设置折线

            Dot3D dot13 = new Dot3D(100, 100,20);

            Dot3D dot23 = new Dot3D(1000, 1000,200);

            varLine3D.Append(dot13);

            varLine3D.Append(dot23);

 

            //圆形管道截面

            PipeCircleSec CircleSec = new PipeCircleSec(25, 3, 1); //参数说明：圆的个数，半径， 厚度

 

            //管线建模信息

            PipeModelInfo info = new PipeModelInfo();

            info.CBType = CornerBuildMode.Direct;//管线拐弯处建模方式--- 直接建模

            info.IsDoubleCreate = false; //是否双面建模

            info.IsShowEdges = false;    //是否显示模型分明的棱角

 

            int rtn = GeoModelCommon.CreatePipeSurByLine(multiSurface, varLine3D, CircleSec, info, -20, null);

 

            SurfaceInfo surfaceInfo = new SurfaceInfo();

            surfaceInfo.FillColor = 6;

            surfaceInfo.LibID = 0;

            surfaceInfo.PatID = 30000032;

            surfaceInfo.TransparentColor = 20;

 

            sfcls.Append(multiSurface, null, surfaceInfo);

            if (sfcls.HasOpen())

            {

                MessageBox.Show("创建管道成功");

                ShowLayer(sfcls, this.sceneCtrl);

            }

            else

                MessageBox.Show("创建管道失败");

            return;

        }
