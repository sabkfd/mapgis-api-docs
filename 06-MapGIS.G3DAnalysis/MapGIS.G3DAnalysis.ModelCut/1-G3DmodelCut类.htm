CutSFClsByGeometry3D(

   SFeatureCls sfcls,

   IGeometry3D cutObj,

   DataBase dataBase,

   int dsID,

   int saveMdl,

   string mdlNamePrefix,

   ref List<int> mdlClsIDs,

   int saveSect,

   string regNamePrefix,

   double dScalY

int

成功返回1，失败返回0

面，多面，体，多体切割面或体简单要素类，支持保存切割后模型，切割剖面等。 

sfcls：被切割的面或体简单要素类

cutObj：切割对象，支持面，多面，体，多体

dataBase：保存切割后模型或剖面的数据库对象，与mdlClsIDs和sectClsIDs中简单要素类一致，如果传空则从sfcls中获取。

dsID：内部自动创建简单要素类时数据集ID，可以传0。

saveMdl：保存切割后模型的标识。大于0，保存模型;等于0，不保存模型。

mdlNamePrefix：内部自动创建简单要素类时模型名称前缀，可以传空。

mdlClsIDs：传入传出切割后模型的简单要素类ID列表，如果不保存模型可以传空。如果需要将模型保存到指定的简单要素类中，则上层接口调用者创建好简单要素类，传入ID列表；也可以传入地理数据库，数据集ID和简单要素类名称前缀，内部自动创建若干个简单要素类。

saveSect：保存切割后剖面的标识，对切割体时有效。大于0，保存剖面；等于0，不保存剖面。

sectNamePrefix：内部自动创建简单要素类时剖面名称前缀，可以传空。

sectClsIDs：传入传出切割后剖面的简单要素类ID列表，如果不保存剖面可以传空。如果需要将剖面保存到指定的简单要素类中，则上层接口调用者创建好简单要素类，传入ID列表；也可以传入地理数据库，数据集ID和简单要素类名称前缀，内部自动创建若干个简单要素类。

saveReg：保存切割后二维剖面的标识，对切割体时有效。大于0，保存二维剖面；等于0，不保存二维剖面。

regNamePrefix：内部自动创建二维剖面简单要素类时二维剖面名称前缀，可以传空。

regClsIDs：传入传出切割后二维剖面的简单要素类ID列表，如果不保存剖面可以传空。如果需要将剖面保存到指定的简单要素类中，则上层接口调用者创建好简单要素类，传入ID列表；也可以传入地理数据库，数据集ID和简单要素类名称前缀，内部自动创建若干个简单要素类。

dScalX：二维剖面横向比例尺

dScalY：二维剖面纵向比例尺

CutGeometryByGeometry 方法示例

C#

using MapGIS.GeoDataBase;

using MapGIS.GeoObjects.Geometry3D;

using MapGIS.G3DAnalysis.ModelCut;

using MapGIS.GeoObjects.Geometry;

using MapGIS.G3DAnalysis.GeoModeling;

 

 Server svr = new Server();

 //连接数据源

 svr.Connect("MapGISLocal", "", "");

 

 DataBase db = svr.OpenGDB("示例数据");

 if (db != null)

 {

     //该示例是以Z轴为切割轴实现圆柱体切割功能

     SFeatureCls srcsfcls = new SFeatureCls(db);

     //打开被切割简单要素类

     bool rtn = srcsfcls.Open("景观_模型", 0);

 

     if (rtn)

     {

         //创建结果简单要素类对象，为剖切模型的结果简单要素类1。其中创建的两个结果类必须保存在数据库同一目录下

         SFeatureCls leftSfcls = new SFeatureCls();

         int leftClsID = leftSfcls.Create("gdbp://MapGisLocal/示例数据/sfcls/景观_模型_圆柱体切割cutA", srcsfcls.GeomType);

 

         //创建结果简单要素类对象，为剖切模型的结果简单要素类2

         SFeatureCls rightSfcls = new SFeatureCls();

         int rightClsID = rightSfcls.Create("gdbp://MapGisLocal/示例数据/sfcls/景观_模型_圆柱体切割cutB", srcsfcls.GeomType);

 

         //创建三维模型切割对象

         G3DModelCut modelCut = new G3DModelCut();

         //构建切割圆柱体对象

         GeoCir cir = new GeoCir();

 

         //设置切割圆柱参数，中心点及圆柱半径

         cir.Set(new Dot(75.4399, 90.382), 50);

         cir.DisperseToDots(10.97 / Math.Min(386, 437) * 20);

         GeoLines lines = new GeoLines();

         lines.Append(cir);

 

         GeoPolygon polygon = new GeoPolygon();

         polygon.Append(lines);

 

         //构建圆柱体

         GeoAnyEntity entity = new GeoAnyEntity();

         GeoModelCommon.CreateClosedSurByReg(entity, polygon, (float)31.5, (float)-5.25, null, 1);

 

         IGeometry3D geomtry = entity;

         if (geomtry != null)

         {

         

             GeomInfo gInfo1 =srcsfcls.GetInfo(15);

             GeomInfo gInfo2 = srcsfcls.GetInfo(15);

             GeoMultiSurface geomA =srcsfcls.GetGeometry(15) as GeoMultiSurface;

             GeoMultiSurface resGeom1=new GeoMultiSurface ();

 

             GeoMultiSurface resGeom2=new GeoMultiSurface ();

             Dot3D refDot=new Dot3D (0,0,0);

             int i = modelCut.CutGeometryByGeometry(geomA, srcsfcls.GetInfo(1), geomtry, refDot, resGeom1, gInfo1, resGeom2, gInfo2);

 

             if (i > 0)

             {

                 leftSfcls.Append(resGeom1, null, gInfo1);

                 rightSfcls.Append(resGeom2,null, gInfo2);

                 MessageBox.Show("切割成功！");

             }

             else

             {

                 MessageBox.Show("切割失败！");

             }

             srcsfcls.Close();

             leftSfcls.Close();

             rightSfcls.Close();

             return;

         }

     }

     else

     {

         MessageBox.Show("打开失败！", "提示");

         return;

     }

 }

 

CutSFCls 方法示例

C#

 

using MapGIS.GeoDataBase;

using MapGIS.G3DAnalysis.ModelCut;

using MapGIS.GeoObjects.Geometry3D;

using MapGIS.GeoObjects.Geometry;

using MapGIS.GeoObjects.Info;

 

//本示例是以YOZ为切割面

 

SFeatureCls sfcls = new SFeatureCls();

//根据url打开被切割简单要素类

bool rtn = sfcls.Open("gdbp://MapGisLocal/TestDB/sfcls/景观_模型");

if (rtn)

{

    //创建结果简单要素类对象，为剖切模型的结果简单要素类1

    SFeatureCls leftSfcls = new SFeatureCls();

    int r = leftSfcls.Create("gdbp://MapGisLocal/TestDB/sfcls/景观_模型_cutA", sfcls.GeomType);

 

    //创建结果简单要素类对象，为剖切模型的结果简单要素类2

    SFeatureCls rightSfcls = new SFeatureCls();

    int rr = rightSfcls.Create("gdbp://MapGisLocal/TestDB/sfcls/景观_模型_cutB", sfcls.GeomType);

 

    if (r > 0 && rr > 0)

    {

        //创建三维模型切割对象

        G3DModelCut modelCut = new G3DModelCut();

 

        //创建几何单面对象

        GeoAnySurface surface = new GeoAnySurface();

        Dots3D dots = new Dots3D();

        dots.Append(new Dot3D(75.43, 9.87, 22.05));

        dots.Append(new Dot3D(75.43, 170.88, 22.05));

        dots.Append(new Dot3D(75.43, 170.88, -1.05));

        dots.Append(new Dot3D(75.43, 9.87, -1.05));

        dots.Append(new Dot3D(75.43, 9.87, -1.05));

 

        uint[] tran = new uint[] { 0, 1, 2, 0, 2, 3 };

        surface.Set(dots, tran, null, null, null, 0, null);

 

        //创建面图形参数对象，为切割结果面的符号对象

        SurfaceInfo surInfo1 = new SurfaceInfo();

        surInfo1.PatID = 30000001;

        surInfo1.FillColor = 2;

        surInfo1.TransparentColor = 0;

        surInfo1.TextureScale = 1;

 

        //创建面图形参数对象，为切割结果面的符号对象

        SurfaceInfo surInfo2 = new SurfaceInfo();

        surInfo2.PatID = 30000001;

        surInfo2.FillColor = 2;

        surInfo2.TransparentColor = 0;

        surInfo2.TextureScale = 1;

 

        //调用批量切割面简单要素类方法，执行切割

        int t = modelCut.CutSFCls(sfcls, surface, null, leftSfcls, surInfo1, rightSfcls, surInfo2，CutType.CutOther);

        if (t > 0)

            MessageBox.Show("切割成功！");

        else

            MessageBox.Show("切割失败！");

        sfcls.Close();

        leftSfcls.Close();

        rightSfcls.Close();

        return;

    }

}

else

{

    MessageBox.Show("打开失败！", "提示");

    return;

}

 

CutSFClsByGeometry3D 方法示例

本示例为实现圆柱体切割功能。

C#

 

using MapGIS.GeoDataBase;

using MapGIS.GeoObjects.Geometry3D;

using MapGIS.G3DAnalysis.ModelCut;

using MapGIS.GeoObjects.Geometry;

using MapGIS.G3DAnalysis.GeoModeling;

 

 

 Server svr = new Server();

 //连接数据源

 svr.Connect("MapGISLocal", "", "");

 

 DataBase db = svr.OpenGDB("示例数据");

 if (db != null)

 {

     //该示例是以Z轴为切割轴实现圆柱体切割功能

     SFeatureCls srcsfcls = new SFeatureCls(db);

     //打开被切割简单要素类

     bool rtn = srcsfcls.Open("景观_模型", 0);

 

     if (rtn)

     {

         //创建结果简单要素类对象，为剖切模型的结果简单要素类1

         SFeatureCls leftSfcls = new SFeatureCls();

         int leftClsID = leftSfcls.Create("gdbp://MapGisLocal/示例数据/sfcls/景观_模型_圆柱体切割cutA", srcsfcls.GeomType);

 

         //创建结果简单要素类对象，为剖切模型的结果简单要素类2

         SFeatureCls rightSfcls = new SFeatureCls();

         int rightClsID = rightSfcls.Create("gdbp://MapGisLocal/示例数据/sfcls/景观_模型_圆柱体切割cutB", srcsfcls.GeomType);

 

         //创建三维模型切割对象

         G3DModelCut modelCut = new G3DModelCut();

 

         //构建切割圆柱体对象

         GeoCir cir = new GeoCir();

         //设置切割圆柱参数，构造圆柱体需要中心点及圆柱半径

         cir.Set(new Dot(75.4399, 90.382), 20);

         cir.DisperseToDots(10.97 / Math.Min(386, 437) * 20);

         GeoLines lines = new GeoLines();

         lines.Append(cir);

 

         GeoPolygon polygon = new GeoPolygon();

         polygon.Append(lines);

 

         //构建圆柱体

         GeoAnyEntity entity = new GeoAnyEntity();

         //根据区创建封闭的面（结果几何实体、范围面、圆柱高、最小Z值、地形数据集、地形高程缩放比）

         GeoModelCommon.CreateClosedSurByReg(entity, polygon, (float)31.5, (float)-5.25, null, 1);

 

         IGeometry3D geomtry = entity;

         Model model = new Model();

         model.FromGeometry(entity);

         model.CreateAnyEntity();

         uint ii = this.sceneCtrl.Append(model);

         this.sceneCtrl.UpdateScene(this.sceneCtrl.GetScene(0), 0, ii, ModelOperType.UpdateMdl);

         if (geomtry != null)

         {

             List<int> cutMdlClsIDs = new List<int>();//初始化三维切割模型简单要素类ID列表

             List<int> cutSectClsIDs = new List<int>();//初始化三维切割剖面简单要素类ID列表

             List<int> cutRegClsIDs = new List<int>(); //初始化二维切割剖面简单要素类ID列表

 

             // 在列表中添加三维切割模型简单要素类ID

             cutMdlClsIDs.Add(leftClsID);

             cutMdlClsIDs.Add(rightClsID);

 

             //执行切割

             //CutSFClsByGeometry3D方法参数说明：

             // SFeatureCls sfcls，被切割的源简单要素类，可以是体模型，也可以是面模型；

             //IGeometry3D cutObj，做切割的实体，如平面、长方体、圆柱体等

             //DataBase dataBase，源类所在的数据库

             //int dsID，源类所在要素数据集的ID

             //int saveMdl，是否保存成切割模型的标志，0/1代表否/是

             //string mdlNamePrefix，三维切割模型的标志名称，随意赋值，无特别意义

             //ref List<int> mdlClsIDs，三维切割模型简单要素类ID列表

             //int saveSect，是否保存成三维剖面的标志，0/1代表否/是,其中只有当被切割的模型类型为体模型时有效

             //string sectNamePrefix，三维剖面的标志名称，随意赋值，无特别意义

             //ref List<int> sectClsIDs，三维切割剖面简单要素类ID列表

             //int saveReg，是否保存成二维维剖面的标志，0/1代表否/是,此参数无效，平台暂时无法实现生成二维剖面的功能

             //string regNamePrefix，二维剖面的标志名称，随意赋值，无特别意义

             //ref List<int> regClsIDs，二维切割剖面简单要素类ID列表

             //double dScalX，横向比例尺

             //double dScalY，纵向比例尺

             int i = modelCut.CutSFClsByGeometry3D(srcsfcls, geomtry, srcsfcls.GDataBase, srcsfcls.DsID, 1, "圆柱体切割模型", ref cutMdlClsIDs, 0, "圆柱体三维切割剖面", ref cutSectClsIDs, 0, "圆柱体二维切割剖面", ref cutRegClsIDs, 1, 1);

 

             if (i > 0)

             {

                 MessageBox.Show("切割成功！");

             }

             else

             {

                 MessageBox.Show("切割失败！");

             }

             srcsfcls.Close();

             return;

         }

     }

     else

     {

         MessageBox.Show("打开失败！", "提示");

         return;

     }

 }
