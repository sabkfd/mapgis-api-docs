示例

CreateAnimFly(Camera,List<AnimFlyNodeInfo>,String,FlyType) 方法示例

C#

using MapGIS.G3DAnalysis.PathFly;     

 using MapGIS.Scene3D;

 using MapGIS.GeoMap;

 using MapGIS.GeoDataBase;

 using MapGIS.G3DAnalysis.GeoModeling;

 using MapGIS.GeoObjects.Geometry;

 

Server svr = new Server();

//连接数据源

svr.Connect("mapgislocal", "", "");

SFeatureCls sfcls = new SFeatureCls();

//根据url打开简单要素类

bool rtn = sfcls.Open("gdbp://MapGisLocal/示例数据/ds/三维示例/sfcls/景观_模型");

if (rtn)

{

    //创建模型图层对象

    ModelLayer modelLayer = new ModelLayer();

    //关联数据

    modelLayer.AttachData(sfcls);

    //创建场景对象

    Scene scene = new Scene();

    //添加模型图层到场景中

    scene.Append(modelLayer);

    //添加场景到场景显示控件中

    this.sceneCtrl.Append(scene);

    //更新场景视图

    this.sceneCtrl.Refresh();

    //复位显示

    this.sceneCtrl.Reset();

 

    //创建相机对象

    Camera camera = new Camera();

    //获取当前场景的相机

    sceneCtrl.GetCamera(ref camera);

    G3DAnimFly g3dAnimFly = new G3DAnimFly();

    //控制点信息列表 

    List<AnimFlyNodeInfo> nodeInfoLst = new List<AnimFlyNodeInfo>();

    //路径操作对象

    G3DPathOperate pathOper = new G3DPathOperate(camera);

    //设置漫游路径坐标点(设置设置三个点）

    Dot3D dot1 = new Dot3D(26.1345386505127, 19.2962131500244, 3.70347356796265);

    Dot3D dot2 = new Dot3D(48.90087890625, 36.9625549316406, 8.543212890625);

    Dot3D dot3 = new Dot3D(61.8438110351563, 53.5401191711426, 9.49627590179443);

    //分别设置三个节点的漫游信息

    AnimFlyNodeInfo nodeInfo1 = new AnimFlyNodeInfo();

    //路径点坐标 

    nodeInfo1.Pos = dot1;

    //设置节点速度 

    nodeInfo1.Speed = 5;

    //路径漫游路径节点信息

    AnimFlyNodeInfo nodeInfo2 = new AnimFlyNodeInfo();

    nodeInfo2.Pos = dot2;

    nodeInfo2.Speed = 5;

    //路径漫游路径节点信息

    AnimFlyNodeInfo nodeInfo3 = new AnimFlyNodeInfo();

    nodeInfo3.Pos = dot3;

    nodeInfo3.Speed = 5;

    //将路径漫游节点信息添加到数组中

    nodeInfoLst.Add(nodeInfo1);

    nodeInfoLst.Add(nodeInfo2);

    nodeInfoLst.Add(nodeInfo3);

 

    //创建插值对象

    InsertValueInfo insertValueInfo = new InsertValueInfo();

    //设置插值类型为线性插值

    insertValueInfo.InsertValueType = InterpolationType.Type_Spline;

    //是否过控制点

    insertValueInfo.IsPassControlPnt = true;

    //开始段比例 

    insertValueInfo.StartScale = 0.2f;

    //结束段比例 

    insertValueInfo.EndScale = 0.2f;

    //光滑度数，范围为1到10,值越大越光滑，仅对样条插值有效 

    insertValueInfo.SmoothValue = PathSmoothType.Smooth_Nice;

    if (pathOper != null)

    {

        //清除绘制

        pathOper.ClearDraw();

        //设置路径信息，

        pathOper.SetPathInfo(nodeInfoLst, false);

        //设置路径漫游插值信息 

        pathOper.SetInsertInfo(insertValueInfo);

        //绘制路径 

        pathOper.DrawPathInfo();

    }

 

    //动画模型名称

    string meshfile = "robot.mesh";

    //动画的动作名称

    string animationName = "Walk";

    //物体缩放比

    double objScaleX = 0.08;

    double objScaleY = 0.08;

    double objScaleZ = 0.06;

    //物体旋转的角度

    double degree = 1;

    //物体旋转的方向

    double directX = 0;

    double directY = 0;

    double directZ = 1.2;

 

    // 漫游前设置合适的飞行模式

    FlyType flytype = this.sceneCtrl.Mode == SceneMode.LOCAL ? FlyType.Type_Plane : FlyType.Type_Spere;

    //创建动画

    bool createFlag = g3dAnimFly.CreateAnimFly(camera, nodeInfoLst, @meshfile, flytype);

备注

运行该示例,MapGIS 10安装路径下（D:\MapGIS 10\Program\media\PathFlyEdit\models）需有robot.mesh文件。
