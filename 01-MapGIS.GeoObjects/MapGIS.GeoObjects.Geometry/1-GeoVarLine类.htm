示例

Mirror 方法示例

C#

using MapGIS.GeoObjects.Geometry;

using MapGIS.GeoMap;

using MapGIS.GISControl;

 

//定义一个MapControl控件对象

MapControl mapControl1 = new MapControl();

mapControl1.Dock = DockStyle.Fill;

this.panel1.Controls.Add(mapControl1); //把MapControl控件承载在Panel控件上面

 

//自定义绘图对象

Display disp = mapControl1.Display;

GeoVarLine varLineA = new GeoVarLine();//折线A

GeoLines linesA = new GeoLines();

GeoVarLine varLineB = new GeoVarLine();//折线B，镜像标准线段

GeoLines linesB = new GeoLines();

GeoVarLine varLineC = new GeoVarLine();//折线C，存储折线A镜像的结果

GeoLines linesC = new GeoLines();

//构造折线A

varLineA.Append(new Dot(50, 50));

varLineA.Append(new Dot(150, 50));

linesA.Append(varLineA);

//构造折线B

varLineB.Append(new Dot(160, 50));

varLineB.Append(new Dot(160, 150));

linesA.Append(varLineB);

 

varLineC=(GeoVarLine)varLineA.Mirror(new Dot(160, 50), new Dot(160, 150));//镜像

linesC.Append(varLineC);

//绘制A

disp.Begin();

disp.PolyPolyLine(linesA);

disp.End();

//绘制C

disp.Begin();

disp.PolyPolyLine(linesC);

disp.End();

 

Resize 方法示例

C#

using MapGIS.GeoObjects.Geometry;

using MapGIS.GeoMap;

using MapGIS.GISControl;

 

//定义一个MapControl控件对象

MapControl mapControl1 = new MapControl();

mapControl1.Dock = DockStyle.Fill;

this.panel1.Controls.Add(mapControl1); //把MapControl控件承载在Panel控件上面

 

//自定义绘图对象

Display disp = mapControl1.Display;

 

GeoVarLine varLineA = new GeoVarLine();

GeoLines linesA = new GeoLines();

GeoVarLine varLineB = new GeoVarLine();

GeoLines linesB = new GeoLines();

//构造折线A

varLineA.Append(new Dot(50, 50));

varLineA.Append(new Dot(100, 50));

varLineA.Append(new Dot(100, 100));

varLineA.Append(new Dot(50, 100));

varLineA.Append(new Dot(50, 50));

linesA.Append(varLineA);

//构造折线B

varLineB = (GeoVarLine)varLineA.Resize(new Rect(0, 0, 200, 200));//缩放

linesB.Append(varLineB);

//绘制A

disp.Begin();

disp.SetPen(2,6);

disp.PolyPolyLine(linesA);

disp.End();

//绘制B

disp.Begin();

TransSRS(SRefData,SRefData) 方法示例

C#

using MapGIS.GeoObjects;

using MapGIS.GeoObjects.Geometry;

using MapGIS.GeoObjects.SpatialRef;

using MapGIS.GeoDataBase;

 

 

//变量定义

Server srv = null;//数据源

DataBase db = null;//数据库

//空间参照系管理对象

SpatialRefMng refMng = null;

//源坐标系

SRefData srsSaref = null;

//目标坐标系

SRefData desSaref = null;

 

//简单要素类

SFeatureCls sfcls = null;

//源折线对象

GeoVarLine varLineA = null;

//目的折线对象

GeoVarLine varLineB = null;

 

//初始化

srv = new Server();

srv.Connect("mapgislocal", "", "");

db = srv.OpenGDB("sample");

           

refMng = new SpatialRefMng(db);

//地理坐标系(北京)度

srsSaref = refMng.Get(2);

//高斯大地坐标系_北京54_36带3_北

desSaref = refMng.Get(51);

 

varLineA = new GeoVarLine();

varLineB = new GeoVarLine();

//构造折线A

varLineA.Append(new Dot(112.875, 28.12));

varLineA.Append(new Dot(113.125, 28.12));

varLineA.Append(new Dot(113.125, 28.25));

varLineA.Append(new Dot(112.875, 28.25));

varLineA.Append(new Dot(112.875, 28.12));

//投影变换

varLineB = (GeoVarLine)varLineA.TransSRS(srsSaref, desSaref);

//为便于查看投影后结果

//将结果存储在简单要素类中

sfcls = (SFeatureCls)db.GetXClass(XClsType.SFCls);

sfcls.Create("srsTrans", GeomType.Lin, 0, 51, null);

sfcls.Append(varLineB, null, null);

//关闭相关对象

sfcls.Close();

db.Close();

srv.DisConnect();
